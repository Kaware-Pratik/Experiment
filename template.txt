# terraform init
- id: 'tf init'
  name: 'hashicorp/terraform:1.0.0'
  entrypoint: 'sh'

  - '-c'
  - |
      terraform init -get=true -upgrade -input=false -backend=true \

      -backend-config="prefix=state/${_STATE_FLD}"

  - name: 'ssh'
